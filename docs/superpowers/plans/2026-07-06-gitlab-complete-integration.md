# GitLab Complete Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Complete the GitLab token-based integration with MR refresh, approvals, discussions, jobs, trace summaries, artifacts, diagnostics, and issue-detail UI while preserving the existing GitHub integration.

**Architecture:** Expand only the GitLab integration path: new `gitlab_*` schema, sqlc queries, GitLab REST client methods, a GitLab-only sync service, GitLab handler endpoints, core schemas/query keys, and GitLab-specific UI. Avoid provider-neutral abstractions and avoid changing GitHub handlers/queries/settings behavior.

**Tech Stack:** Go 1.26 backend with Chi, pgx/sqlc, PostgreSQL migrations, TypeScript strict mode, Zod API parsing, TanStack Query, React shared views, Vitest, Go tests.

func TestListPipelineJobsFollowsPagination(t *testing.T) {
	var paths []string
	srv := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		paths = append(paths, r.URL.String())
		w.Header().Set("Content-Type", "application/json")
		switch r.URL.Query().Get("page") {
		case "", "1":
			w.Header().Set("X-Next-Page", "2")
			_, _ = w.Write([]byte(`[{"id":11,"name":"test","status":"failed","stage":"test","web_url":"https://gitlab/jobs/11"}]`))
		case "2":
			_, _ = w.Write([]byte(`[{"id":12,"name":"lint","status":"success","stage":"test","web_url":"https://gitlab/jobs/12"}]`))
		default:
			t.Fatalf("unexpected page %q", r.URL.Query().Get("page"))
		}
	}))
	defer srv.Close()
	c, err := NewClient(Config{BaseURL: srv.URL, Token: "token"})
	if err != nil {
		t.Fatalf("NewClient: %v", err)
	}
	jobs, err := c.ListPipelineJobs(context.Background(), 42, 77)
	if err != nil {
		t.Fatalf("ListPipelineJobs: %v", err)
	}
	if len(jobs) != 2 || jobs[0].ID != 11 || jobs[1].ID != 12 {

	}
	if len(paths) != 2 || !strings.Contains(paths[1], "page=2") {
		t.Fatalf("paths = %+v", paths)
	}
}

const defaultMaxPages = 10

func (c *Client) doPagedJSON(ctx context.Context, path string, out any) error {
	page := 1
	values := reflect.ValueOf(out)
	if values.Kind() != reflect.Pointer || values.Elem().Kind() != reflect.Slice {
		return errors.New("gitlab paged output must be pointer to slice")
	}
	for {
		pagePath := path
		sep := "?"
		if strings.Contains(pagePath, "?") {
			sep = "&"
		}
		pagePath += fmt.Sprintf("%spage=%d&per_page=100", sep, page)
		var pageOut = reflect.New(values.Elem().Type()).Interface()
		resp, err := c.doJSONResponse(ctx, http.MethodGet, pagePath, nil, pageOut)
		if err != nil {
			return err
		}
		values.Elem().Set(reflect.AppendSlice(values.Elem(), reflect.ValueOf(pageOut).Elem()))
		next := strings.TrimSpace(resp.Header.Get("X-Next-Page"))
		if next == "" {
			return nil
		}
		page++
		if page > defaultMaxPages {
			return fmt.Errorf("gitlab pagination limit exceeded for %s", path)
		}
	}
}

	issue, ok := h.loadIssueForUser(w, r, chi.URLParam(r, "id"))
	if !ok {
		return
	}
	mrUUID, ok := parseUUIDOrBadRequest(w, chi.URLParam(r, "mrId"), "merge request id")
	if !ok {
		return
	}
	mr, err := h.Queries.GetGitLabMergeRequestForIssue(r.Context(), db.GetGitLabMergeRequestForIssueParams{
		IssueID: issue.ID,
		ID: mrUUID,
		WorkspaceID: issue.WorkspaceID,
	})
	if err != nil {
		writeError(w, http.StatusNotFound, "gitlab merge request not found")
		return
	}
	binding, err := h.Queries.GetGitLabProjectBinding(r.Context(), db.GetGitLabProjectBindingParams{
		ID: mr.ProjectBindingID,
		WorkspaceID: issue.WorkspaceID,
	})
	if err != nil {
		writeError(w, http.StatusNotFound, "gitlab project not found")
		return
	}
	cfg := gitlab.LoadConfigFromEnv()
	api, err := newGitLabClient(cfg)
	if err != nil {
		writeError(w, http.StatusBadGateway, err.Error())
		return
	}
	detailAPI, ok := api.(gitLabDetailAPI)
	if !ok {
		writeError(w, http.StatusBadGateway, "gitlab client does not support details refresh")
		return
	}
	svc := newGitLabSyncService(h, detailAPI)
	refreshed, err := svc.refreshMergeRequest(r.Context(), binding, mr.ID)
	if err != nil {
		writeError(w, http.StatusBadGateway, err.Error())
		return
	}
	writeJSON(w, http.StatusOK, h.buildGitLabMRDetailsResponse(r.Context(), issue.WorkspaceID, refreshed))

func (h *Handler) RefreshGitLabProjectBinding(w http.ResponseWriter, r *http.Request) {
	workspaceID := chi.URLParam(r, "id")
	wsUUID, ok := parseUUIDOrBadRequest(w, workspaceID, "workspace id")
	if !ok {
		return
	}
	bindingUUID, ok := parseUUIDOrBadRequest(w, chi.URLParam(r, "bindingId"), "binding id")

		WorkspaceID: wsUUID,
	})
	if err != nil {
		writeError(w, http.StatusNotFound, "gitlab project not found")
		return
	}
	_ = h.Queries.MarkGitLabProjectRefreshStarted(r.Context(), db.MarkGitLabProjectRefreshStartedParams{ID: binding.ID, WorkspaceID: wsUUID})
	cfg := gitlab.LoadConfigFromEnv()
	api, err := newGitLabClient(cfg)
	if err != nil {
		_ = h.Queries.MarkGitLabProjectRefreshFinished(r.Context(), db.MarkGitLabProjectRefreshFinishedParams{ID: binding.ID, WorkspaceID: wsUUID, LastRefreshError: strToText(err.Error())})
